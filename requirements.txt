google-genimport streamlit as st
import google.generativeai as genai
import glob
import smtplib
import json
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from streamlit_local_storage import LocalStorage

# --- INITIALIZE BROWSER LOCAL STORAGE ---
local_storage = LocalStorage()

st.set_page_config(page_title="Meu Tutor de Português", page_icon="🇵🇹", layout="centered")

# --- BACKUP LOGIC: RESCUE CHAT & CREDENTIALS FROM PHONE STORAGE ---
saved_api_key = local_storage.getItem("gemini_api_key") or ""
saved_user_email = local_storage.getItem("user_email") or ""
saved_sender_email = local_storage.getItem("sender_email") or ""
saved_sender_pwd = local_storage.getItem("sender_password") or ""
saved_history_json = local_storage.getItem("chat_history") or "[]"

# Recover active session history if it vanished from short-term memory
if "messages" not in st.session_state:
    try:
        st.session_state.messages = json.loads(saved_history_json)
    except:
        st.session_state.messages = []

if "email_sent" not in st.session_state:
    st.session_state.email_sent = False

st.title("🇵🇹 Persistent AI Portuguese Tutor")
st.write("Your credentials and conversation history are now auto-saved directly on your phone!")

# --- GLITCH 1 FIX: VISUAL PROMPT ALERT FOR UNSAVED TRANSCRIPTS ---
if st.session_state.messages and not st.session_state.email_sent:
    st.warning("⚠️ **Unsaved Drill Session Active!** Remember to tap the blue 'Email Me' button at the bottom before starting a brand-new module so you don't lose your progress.")
elif st.session_state.email_sent:
    st.success("✅ Current chat transcript has been securely backed up via email. It is safe to clear or close!")

st.markdown("---")

# --- SIDEBAR CONFIGURATION (PRE-FILLED FROM BROWSER COOKIES) ---
st.sidebar.title("Configuração")
api_key = st.sidebar.text_input("Enter Gemini API Key:", value=saved_api_key, type="password")

# Auto-save API Key to phone if it changes
if api_key != saved_api_key:
    local_storage.setItem("gemini_api_key", api_key)

st.sidebar.markdown("---")
st.sidebar.subheader("📬 Email Transcript Setup")
user_email = st.sidebar.text_input("Your Email Address (Recipient):", value=saved_user_email)
sender_email = st.sidebar.text_input("Sender Email Address:", value=saved_sender_email)
sender_password = st.sidebar.text_input("Sender App Password:", value=saved_sender_pwd, type="password")
smtp_server = st.sidebar.selectbox("Email Provider:", ["smtp.gmail.com", "smtp.mail.yahoo.com", "smtp-mail.outlook.com"])

# Save email parameters dynamically to phone storage
if user_email != saved_user_email: local_storage.setItem("user_email", user_email)
if sender_email != saved_sender_email: local_storage.setItem("sender_email", sender_email)
if sender_password != saved_sender_pwd: local_storage.setItem("sender_password", sender_password)

# Clear Button to manually reset chat history when user wishes to start fresh
if st.sidebar.button("🗑️ Clear and Start Fresh Session"):
    st.session_state.messages = []
    st.session_state.email_sent = False
    local_storage.setItem("chat_history", "[]")
    st.rerun()

if not api_key:
    st.info("💡 Enter your Gemini API Key in the sidebar to begin.")
else:
    genai.configure(api_key=api_key)
    
    knowledge_base = ""
    txt_files = glob.glob("*.txt")
    if txt_files:
        for file in txt_files:
            with open(file, "r", encoding="utf-8") as f:
                knowledge_base += f"\n--- Source File: {file} ---\n" + f.read()
    else:
        knowledge_base = "Use high-quality European Portuguese rules focused on daily life, food, and culture in Portugal."

    # --- DISPLAY CHAT INTERFACE ---
    for message in st.session_state.messages:
        with st.chat_message(message["role"]):
            st.markdown(message["content"])

    if user_prompt := st.chat_input("Ask your tutor anything..."):
        with st.chat_message("user"):
            st.markdown(user_prompt)
        st.session_state.messages.append({"role": "user", "content": user_prompt})
        
        # Instantly back up history array to browser storage on every new sentence
        local_storage.setItem("chat_history", json.dumps(st.session_state.messages))
        st.session_state.email_sent = False # New message added, requires fresh email backup

        system_instruction = (
            f"You are an expert Portuguese language tutor teaching an English speaker. "
            f"CRITICAL RULE: Explain grammar entirely in ENGLISH. "
            f"Focus all scenarios on daily life, navigation, traditions, and food in Portugal."
        )

        try:
            model = genai.GenerativeModel('gemini-2.5-flash')
            full_prompt = f"{system_instruction}\n\nUser Request: {user_prompt}"
            with st.chat_message("assistant"):
                response_placeholder = st.empty()
                response = model.generate_content(full_prompt)
                response_placeholder.markdown(response.text)
            st.session_state.messages.append({"role": "assistant", "content": response.text})
            
            # Back up updated history including the AI response
            local_storage.setItem("chat_history", json.dumps(st.session_state.messages))
            st.rerun()
        except Exception as e:
            st.error(f"AI Communication Error: {e}")

    # --- EMAIL EXPORT MODULE ---
    if st.session_state.messages:
        st.markdown("---")
        if st.button("📧 Email Me This Session's Transcript"):
            if not user_email or not sender_email or not sender_password:
                st.warning("⚠️ Please complete your email setup details in the sidebar first!")
            else:
                with st.spinner("Compiling and sending transcript..."):
                    email_body = "Here is the transcript of your Portuguese learning session:\n\n"
                    for msg in st.session_state.messages:
                        speaker = "You" if msg["role"] == "user" else "AI Tutor"
                        email_body += f"[{speaker}]: {msg['content']}\n\n"
                        email_body += "-"*40 + "\n\n"

                    try:
                        msg = MIMEMultipart()
                        msg['From'] = sender_email
                        msg['To'] = user_email
                        msg['Subject'] = "🇵🇹 My Portuguese Session Transcript"
                        msg.attach(MIMEText(email_body, 'plain'))

                        server = smtplib.SMTP_SSL(smtp_server, 465)
                        server.login(sender_email, sender_password)
                        server.sendmail(sender_email, user_email, msg.as_string())
                        server.quit()
                        
                        st.session_state.email_sent = True
                        st.rerun()
                    except Exception as email_err:
                        st.error(f"Failed to send email: {email_err}")erativeai
